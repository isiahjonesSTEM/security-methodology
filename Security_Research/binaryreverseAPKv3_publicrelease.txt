# --- Helper Functions ---
def run_cmd(cmd_list, capture_output=True, stdout=None):
    """Run a command via PowerShell (for binwalk)."""
    cmd_str = ' '.join([f'"{arg}"' if ' ' in arg else arg for arg in cmd_list])
    ps_cmd = ["powershell.exe", "-Command", cmd_str]
    try:
        if stdout:
            result = subprocess.run(ps_cmd, stdout=stdout, stderr=subprocess.PIPE, text=True, check=False)
            return result.stderr
        else:
            result = subprocess.run(ps_cmd, capture_output=capture_output, text=True, check=False)
            return result.stdout if capture_output else result
    except Exception as e:
        print(f"[!] PowerShell Command failed: {cmd_str}\nError: {e}")
        return ""

def get_file_type_win(path):
    ext = Path(path).suffix.lower()
    if ext == '.apk': return 'android package'
    elif ext in ['.exe', '.dll', '.bin', '.elf', '.img', '.ubi', '.squashfs']: return 'binary/firmware'
    else:
        output = run_cmd(['file', path])
        return output.lower() if output else 'unknown'

def collect_extracted_files(extracted_dir):
    extracted_path = Path(extracted_dir)
    if not extracted_path.exists(): return []
    files = []
    for root, dirs, filenames in os.walk(extracted_path):
        for f in filenames:
            rel_path = os.path.relpath(os.path.join(root, f), extracted_path)
            files.append(rel_path)
    return files

def scrape_endpoints(search_path):
    """Scrape IPs and URLs from all generated output folders."""
    url_pattern = r'https?://(?:[-\w.]|(?:%[\da-fA-F]{2}))+[/\w\.-]*'
    ip_pattern = r'\b(?:\d{1,3}\.){3}\d{1,3}\b'
    endpoints = {"urls": set(), "ips": set()}
    for root, _, files in os.walk(search_path):
        for file in files:
            if file.endswith((".java", ".xml", ".txt", ".so", ".json")):
                try:
                    with open(os.path.join(root, file), 'r', encoding='utf-8', errors='ignore') as f:
                        content = f.read()
                        endpoints["urls"].update(re.findall(url_pattern, content))
                        endpoints["ips"].update(re.findall(ip_pattern, content))
                except: continue
    return {"urls": list(endpoints["urls"])[:100], "ips": list(endpoints["ips"])[:50]}

# --- Binwalk Risk Analysis ---
def analyze_binwalk_risks(extract_dir):
    """Identify files that could be used in exploits (e.g., world‑writable .so, path traversal artifacts)."""
    findings = []
    for root, _, files in os.walk(extract_dir):
        for file in files:
            full_path = os.path.join(root, file)
            rel_path = os.path.relpath(full_path, extract_dir)
            # Check for world‑writable .so files (permission 0o002 = others write)
            if file.endswith('.so') and (os.stat(full_path).st_mode & 0o002):
                findings.append({
                    "type": "World_writable_native_library",
                    "path": rel_path,
                    "risk": "Attacker could replace this .so with a malicious one to achieve ACE"
                })
            # Check for extracted path traversal artifacts (indicative of zip slip)
            if '..' in rel_path or any(part == '..' for part in Path(rel_path).parts):
                findings.append({
                    "type": "Path_traversal_artifact",
                    "path": rel_path,
                    "risk": "Indicates possible zip slip vulnerability during extraction"
                })
    return findings

# --- Direct extraction of native libraries from APK (flattened to avoid path issues) ---
def extract_native_libs(apk_path, out_dir):
    """Extract all .so files from the APK into native_libs/ directory with flattened names."""
    lib_dir = os.path.join(out_dir, "native_libs")
    os.makedirs(lib_dir, exist_ok=True)
    extracted = []
    try:
        with zipfile.ZipFile(apk_path, 'r') as z:
            for file in z.namelist():
                if file.startswith('lib/') and file.endswith('.so'):
                    # Flatten the path: replace slashes with underscores to keep arch info
                    flat_name = file.replace('/', '_').replace('\\', '_')
                    target = os.path.join(lib_dir, flat_name)
                    # Check path length and warn if too long
                    if len(target) > 250:
                        print(f"[!] Warning: Path length {len(target)} may cause issues: {target}")
                    with z.open(file) as source, open(target, 'wb') as dest:
                        shutil.copyfileobj(source, dest)
                    extracted.append(target)
        print(f"[+] Extracted {len(extracted)} native libraries directly from APK.")
    except Exception as e:
        print(f"[!] Failed to extract native libs from APK: {e}")
        import traceback
        traceback.print_exc()
    return extracted

def scan_manifest(source_dir):
    """Identify APK entry points from AndroidManifest.xml."""
    manifest_path = os.path.join(source_dir, "resources", "AndroidManifest.xml")
    entry_points = []
    if os.path.exists(manifest_path):
        with open(manifest_path, 'r', encoding='utf-8', errors='ignore') as f:
            content = f.read()
            if 'android:exported="true"' in content:
                entry_points.append("Exported Components Found")
            schemes = re.findall(r'android:scheme="([^"]+)"', content)
            for s in schemes:
                entry_points.append(f"Deep Link: {s}://")
    return entry_points

def run_static_scanner(search_path, patterns_dict):
    """Generic scanner that returns list of findings with type and file."""
    findings = []
    for root, _, files in os.walk(search_path):
        for file in files:
            if file.endswith((".java", ".kt", ".xml", ".txt")):
                try:
                    with open(os.path.join(root, file), 'r', encoding='utf-8', errors='ignore') as f:
                        content = f.read()
                        for v_type, pattern in patterns_dict.items():
                            if re.search(pattern, content, re.IGNORECASE):
                                findings.append({
                                    "type": v_type,
                                    "file": os.path.relpath(os.path.join(root, file), search_path)
                                })
                except:
                    continue
    return findings

# --- Ghidra / pyghidra functions with thread‑safe initialisation ---
def run_pyghidra(bin_path, out_file):
    """Decompile a binary with pyghidra. Thread‑safe JVM initialisation."""
    if 'GHIDRA_INSTALL_DIR' not in os.environ:
        print("[!] GHIDRA_INSTALL_DIR not set, skipping pyghidra")
        return False

    try:
        import pyghidra
    except ImportError as e:
        print(f"[!] pyghidra import failed: {e}")
        return False

    os.makedirs(os.path.dirname(out_file), exist_ok=True)
    print(f"[+] Opening binary with pyghidra: {bin_path}")

    global _pyghidra_initialised
    with _pyghidra_init_lock:
        if not _pyghidra_initialised:
            try:
                # Force JVM start and pre‑load Ghidra classes using the first binary
                with pyghidra.open_program(bin_path, analyze=True) as _dummy:
                    from ghidra.framework import Application
                    _pyghidra_initialised = True
                    print("[+] PyGhidra JVM initialised and Ghidra classes loaded.")
            except Exception as e:
                print(f"[!] Failed to initialise PyGhidra JVM: {e}")
                return False

    # Actual analysis (JVM already running)
    try:
        with pyghidra.open_program(bin_path, analyze=True) as flat_api:
            program = flat_api.getCurrentProgram()
            if program is None:
                print("[!] No program loaded.")
                return False
            from ghidra.app.decompiler.flatapi import FlatDecompilerAPI
            flat_decompiler = FlatDecompilerAPI(flat_api)
            fm = program.getFunctionManager()
            functions = list(fm.getFunctions(True))
            total_funcs = len(functions)
            print(f"[+] Found {total_funcs} functions to decompile")

            with open(out_file, "w", encoding="utf-8") as f:
                for i, func in enumerate(functions, 1):
                    f.write(f"=== {func.getName()} ===\n")
                    if i % 100 == 0:
                        print(f"[+] Decompiled {i}/{total_funcs} functions...")
                    try:
                        decompiled = flat_decompiler.decompile(func)
                        f.write(str(decompiled) if decompiled else "// Decompilation returned empty result\n")
                    except Exception as e:
                        f.write(f"// Decompilation error: {type(e).__name__}: {e}\n")
                    f.write("\n\n")
            flat_decompiler.dispose()
            print(f"[+] Decompilation complete. Processed {total_funcs} functions.")
            return True
    except Exception as e:
        print(f"[!] pyghidra error for {bin_path}: {e}")
        import traceback
        traceback.print_exc()
        return False

def analyze_native_library(so_path, out_dir):
    """
    Run Ghidra on a single .so, then parse the output to extract structured findings.
    Returns path to a JSON file containing the findings.
    """
    base = os.path.basename(so_path)
    safe_name = base.replace('.', '_')
    decomp_file = os.path.join(out_dir, f"ghidra_{safe_name}.txt")
    success = run_pyghidra(so_path, decomp_file)
    if not success:
        return None

    # Parse the decompiled output for vulnerability patterns
    findings = []
    try:
        with open(decomp_file, 'r', encoding='utf-8') as f:
            content = f.read()
        function_blocks = re.split(r'=== (.*?) ===\n', content)
        for i in range(1, len(function_blocks)-1, 2):
            func_name = function_blocks[i].strip()
            func_body = function_blocks[i+1]
            for v_type, pattern in VULN_PATTERNS.items():
                for match in re.finditer(pattern, func_body, re.IGNORECASE):
                    start = max(0, match.start() - 50)
                    end = min(len(func_body), match.end() + 50)
                    snippet = func_body[start:end]
                    line_no = func_body.count('\n', 0, match.start()) + 1
                    findings.append({
                        "type": v_type,
                        "function": func_name,
                        "line": line_no,
                        "snippet": snippet,
                        "root_cause": f"Use of {v_type} pattern without proper safeguards",
                        "suggested_fix": "Add bounds checking, use safe alternatives, or validate input length.",
                        "patch_example": "if (user_len < sizeof(buffer)) { memcpy(buffer, user_input, user_len); } else { return -1; }"
                    })
    except Exception as e:
        print(f"[!] Error parsing Ghidra output for {so_path}: {e}")

    findings_file = os.path.join(out_dir, f"ghidra_findings_{safe_name}.json")
